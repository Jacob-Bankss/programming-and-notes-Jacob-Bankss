3    import sys

5    from conf import SPEED_LIMITS, VEHICLE_TYPES

7    
8    def initialise_speed_lists():
9        return {vehicle_type: [] for vehicle_type in VEHICLE_TYPES}
10   
11   
12   def parse_speed_line(speed_line):
13       speed_line = speed_line.strip()
14   
15       return speed_line[-1], int(speed_line[:-1])
16   
17   
18   def analyse_the_camera_file(camera_file):
19       speeds = initialise_speed_lists()
20   
21       with open(camera_file, "r") as cf:
22           for speed_line in cf:
23               if speed_line != "OFF":
24                   vehicle_type, speed = parse_speed_line(speed_line)
25                   speeds[vehicle_type].append(speed)
26   
27       analyse_and_print_results(speeds)
28

30   def count_total_vehicles(speeds):
31       return sum([len(speeds[c]) for c in VEHICLE_TYPES])
32   
33   
34   def get_vehicle_type_percentage(speeds, vehicle_type):
35       return len(speeds[vehicle_type]) / count_total_vehicles(speeds) * 100
36   
37   
38   def count_vehicle_type(speeds, vehicle_type):
39       return len(speeds[vehicle_type])
40   
41   
42   def get_vehicle_type_average_speed(speeds, vehicle_type):
43       from statistics import mean
44   
45       return mean(speeds[vehicle_type])
46   
47   
48   def count_vehicle_type_speeders(speeds, vehicle_type, limit):
49       return len([s for s in speeds[vehicle_type] if s > limit])
50   
51   
52   def get_total_speeder_percent(speeds):
53       total_vehicles = count_total_vehicles(speeds)
54       total_speeders = (
55           count_vehicle_type_speeders(speeds, "C", SPEED_LIMITS["C"])
56           + count_vehicle_type_speeders(speeds, "V", SPEED_LIMITS["V"])
57           + count_vehicle_type_speeders(speeds, "B", SPEED_LIMITS["B"])
58           + count_vehicle_type_speeders(speeds, "H", SPEED_LIMITS["H"])
59       )
60   
61       return total_speeders / total_vehicles * 100
62   
63   
64   def print_analysis_header():
65       print()
66       print("Speed Camera Summary")
67       print("====================")
68       print()
69   
70   
71   def analyse_and_print_results(speeds):
72       print_analysis_header()
73       print(f"Total Vehicles Seen: {count_total_vehicles(speeds)}")
74       print()
75       print(f"Percentage Cars:  {get_vehicle_type_percentage(speeds, 'C'):6.2f}%")
76       print(f"Percentage Vans:  {get_vehicle_type_percentage(speeds, 'V'):6.2f}%")
77       print(f"Percentage Buses: {get_vehicle_type_percentage(speeds, 'B'):6.2f}%")
78       print(f"Percentage HGVs:  {get_vehicle_type_percentage(speeds, 'H'):6.2f}%")
79       print()
80       print(f"Average Car Speed: {get_vehicle_type_average_speed(speeds, 'C'):6.2f} MPH")
81       print(f"Average Van Speed: {get_vehicle_type_average_speed(speeds, 'V'):6.2f} MPH")
82       print(f"Average Bus Speed: {get_vehicle_type_average_speed(speeds, 'B'):6.2f} MPH")
83       print(f"Average HGV Speed: {get_vehicle_type_average_speed(speeds, 'H'):6.2f} MPH")
84       print()
85       print(
86           f"Speeding Cars:  {count_vehicle_type_speeders(speeds, 'C', SPEED_LIMITS['C'])}"
87           f" of {count_vehicle_type(speeds, 'C')}"
88       )
89       print(
90           f"Speeding Vans:  {count_vehicle_type_speeders(speeds, 'V', SPEED_LIMITS['V'])}"
91           f" of {count_vehicle_type(speeds, 'V')}"
92       )
93       print(
94           f"Speeding Buses: {count_vehicle_type_speeders(speeds, 'B', SPEED_LIMITS['B'])}"
95           f" of {count_vehicle_type(speeds, 'B')}"
96       )
97       print(
98           f"Speeding HGVs:  {count_vehicle_type_speeders(speeds, 'H', SPEED_LIMITS['H'])}"
99           f" of {count_vehicle_type(speeds, 'H')}"
100      )
101      print()
102      print(f"Overall percentage of speeders: {get_total_speeder_percent(speeds):.2f}%")
