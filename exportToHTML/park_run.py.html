4    def time_as_string(seconds):
5        mins = int(seconds // 60)
6        secs = round(seconds % 60, 1)
7    
8        return f'{mins} minute{"" if mins == 1 else "s"}, {secs} second{"" if secs == 1 else "s"}'

11   def parse_time_record(line_from_the_stream):
12       try:
13           singlet_number, time = line_from_the_stream.split("::")
14   
15           if len(singlet_number) < 1:
16               raise ValueError("Missing singlet number")
17   
18           return singlet_number, int(time)
19       except ValueError:
20           raise ValueError("Error in data. Probably cannot convert time")

48   def process_the_park_run():
49       times = []
50       top_runner = ""
51   
52       while True:
53           next_line = input("> ")
54   
55           if next_line == "END":
56               print_results(times, top_runner)

59               try:
60                   runner, time = parse_time_record(next_line)

62                   if times:

64                   else:
65                       top_runner = runner
66   
67                   times.append(time)
68   
69               except ValueError:
70                   print("Error in data stream. Ignoring. Carry on.")
