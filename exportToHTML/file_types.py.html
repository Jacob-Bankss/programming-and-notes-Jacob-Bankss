4    def get_files_in_current_directory():
5        from os import listdir
6        from os.path import isfile
7    
8        return [f for f in listdir(".") if isfile(f)]

11   def guess_file_type_from_extension(file_name):
12       from os.path import splitext
13   
14       known_types = {
15           "c": "C Program",
16           "cpp": "C++ Program",
17           "json": "JSON Data",
18           "md": "Markdown Document",
19           "mmd": "Mermaid Diagram",
20           "py": "Python Program",
21           "pyc": "Python Byte Code",
22       }
23   
24       _, extension = splitext(file_name)
25   
26       return known_types[extension[1:]]
27   
28   
29   if __name__ == "__main__":
30       for file in sorted(get_files_in_current_directory()):
31           try:
32               print(f"- {file}: {guess_file_type_from_extension(file)}")
33           except KeyError:
34               print(f"- {file}: Unknown")
