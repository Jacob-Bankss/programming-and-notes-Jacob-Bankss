9    def print_user_table(user_list):
10       clean_list = [
11           [
12               "uid",
13               "Username",
14               "Name",
15               "Password",
16               "Admin",
17           ],
18       ]
19   
20       for user in user_list:
21           uid, username, name, _, admin = user
22           clean_list.append(
23               ([
24                   uid,
25                   username,
26                   name,
27                   "#####",
28                   "Yes" if admin else "No",
29               ])
30           )
31   
32       print(tabulate(clean_list, headers="firstrow", tablefmt="grid"))
