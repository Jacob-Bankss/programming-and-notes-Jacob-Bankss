11   SETTINGS_FILE = "settings.json"

14   def check_settings_file():
15       try:
16           json.load(open(SETTINGS_FILE, "r"))["ExitWords"]
17       except FileNotFoundError:
18           raise FileNotFoundError("Settings file not found.")
19       except json.decoder.JSONDecodeError:
20           raise RuntimeError("Settings file is not a valid JSON file.")

23   def start_log_file(logfile):
24       write_to_log(logfile, "SYSTEM", "Chat Started")
25   
26   
27   def log_users(logfile, user_name, agent_name):
28       write_to_log(logfile, "SYSTEM", f"User identified as {user_name}")
29       write_to_log(logfile, "SYSTEM", f"Agent identified as {agent_name}")

32   def write_to_log(logfile, actor, message):
33       from datetime import datetime
34   
35       with open(logfile, "a") as log:
36           log.write(f'{datetime.now().strftime("%H:%M")} --> {actor:<18}: {message}\n')

39   def get_words_from_question(the_question):
40       return remove_punctuation(the_question).lower().split()
41   
42   
43   def get_user_name():
44       while True:
45           name = input("How shall we call you? ")
46   
47           if name:
48               return name.strip().title()
49           else:
50               print("You must enter your name.")
51   
52   
53   def generate_agent_name():
54       possible_names = json.load(open(SETTINGS_FILE, "r"))["OperatorNames"]
55   
56       return choice(possible_names)
57   
58   
59   def user_wants_to_exit(the_question):
60       exit_words = json.load(open(SETTINGS_FILE, "r"))["ExitWords"]
61       the_question = remove_punctuation(the_question)
62   
63       return any(word == the_question for word in exit_words)
64   
65   
66   def generate_response(the_question, name):
67       responses = json.load(open(SETTINGS_FILE, "r"))["Responses"][0]
68   
69       question_words = get_words_from_question(the_question)
70   
71       for keyword in responses.keys():
72           if keyword in question_words:
73               return responses[keyword].replace("{name}", name)
74   
75       return generate_random_response(name)
76   
77   
78   def generate_random_response(name):
79       possible_answers = json.load(open(SETTINGS_FILE, "r"))["DefaultAnswers"]
80   
81       random_answer = choice(possible_answers)
82   
83       return random_answer.replace("{name}", name)

86   def print_friendly_greeting(user_name, agent_name):
87       banner_print("Welcome to LBU Applicant Chat!")
88       print(f"Greetings, {user_name}. I am {agent_name}. How may I serve you? ")
89   
90   
91   def respond_to_questions(user_name, agent_name, logfile):
92       while True:
93           question = input("--> ")
94   
95           if question:
96               if user_wants_to_exit(question):
97                   break
98   
99               write_to_log(logfile, f"User: {user_name}", question)
100  
101              response = generate_response(question, user_name)
102  
103              print(response)
104              write_to_log(logfile, f"Agent: {agent_name}", response)
105

107  def say_farewell(user_name, logfile):
108      banner_print(f"Thanks for using LBU Chat, {user_name}!")
109      write_to_log(logfile, "SYSTEM", "Chat Ended")
110  
111      banner_print(f'A log has been written to "{logfile}".')
112

114  def run_chatbot():
115      logfile = generate_log_file_name_from_time()
116      start_log_file(logfile)
117  
118      user_name, agent_name = get_user_name(), generate_agent_name()
119      log_users(logfile, user_name, agent_name)
120  
121      print_friendly_greeting(user_name, agent_name)
122      respond_to_questions(user_name, agent_name, logfile)
123      say_farewell(user_name, logfile)

126  def be_the_lbu_chatbot():
127      try:
128          check_settings_file()
129      except FileNotFoundError:
130          print("ERROR: Settings file not found. Exiting...")
131      except RuntimeError:
132          print("ERROR: Cannot start, Most likely a JSON problem in Settings. Exiting...")
133      else:
134          run_chatbot()
