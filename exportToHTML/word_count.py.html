1    #!/usr/bin/env python3

3    import sys

5    
6    def read_file(filename):
7        with open(filename) as f:
8            return f.read()
9    
10   
11   def clean_string(s):
12       return "".join([c for c in s if c.isalpha() or c.isspace()])
13   
14   
15   def join_lines(s):
16       return s.replace("\n", " ")
17   
18   
19   def build_word_list(string_of_words):
20       return string_of_words.split()
21   
22   
23   def print_top_words(word_list):
24       from collections import Counter
25   
26       word_frequencies = Counter(word_list)
27       for word, frequency in sorted(
28           word_frequencies.most_common(10), key=lambda x: x[1], reverse=True
29       ):
30           print(f"{word}: {frequency}")
31   
32   
33   def do_the_word_count():
34       try:
35           raw_file = read_file(sys.argv[1])
36       except IndexError:
37           print(f"{sys.argv[0]}: Usage: python3 word_count.py <file>")
38       except FileNotFoundError:
39           print(f'{sys.argv[0]}: Cannot open "{sys.argv[1]}"')
40       else:
41           clean_file = join_lines(clean_string(raw_file))
42           word_list = build_word_list(clean_file)
43           print_top_words(word_list)
44   
45   
46   if __name__ == "__main__":
47       do_the_word_count()
