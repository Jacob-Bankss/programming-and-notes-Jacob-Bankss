generate_random_lap_times.py 
1    #!/usr/bin/env python3
2    
3    from conf import DRIVER_FILE
4    
5    MAX_LAPS = 60
6    MIN_LAPS = 30
7    
8    MAX_LAP_TIME = 120
9    MIN_LAP_TIME = 110
10   
11   TOP_TEAM = [
12       "McLaren",
13       "Ferrari",
14   ]
15   MIDDLE_TEAM = [
16       "Mercedes",
17       "Red Bull",
18       "Williams",
19       "Kick Sauber",
20   ]
21   
22   
23   OUTPUT_FILE = "lap_times.txt"
24   
25   
26   def get_drivers_from_file(driver_file=DRIVER_FILE):
27       drivers = []
28   
29       with open(driver_file, "r") as df:
30           for driver in df:
31               _, short_code, __, team = driver.strip().split(",")
32               drivers.append((short_code, team))
33   
34       return drivers
35   
36   
37   def generate_the_lap_times(driver_file=DRIVER_FILE, lap_file=OUTPUT_FILE):
38       from random import randint, shuffle, uniform
39   
40       drivers = get_drivers_from_file(driver_file)
41   
42       lap_times = []
43   
44       for driver in drivers:
45           code, team = driver
46           for lap in range(randint(MIN_LAPS, MAX_LAPS)):
47               if team in TOP_TEAM:
48                   min_lap_time = MIN_LAP_TIME - 1.5
49                   max_lap_time = MAX_LAP_TIME - 2
50               elif team in MIDDLE_TEAM:
51                   min_lap_time = MIN_LAP_TIME - 1
52                   max_lap_time = MAX_LAP_TIME + 1
53               else:
54                   min_lap_time = MIN_LAP_TIME
55                   max_lap_time = MAX_LAP_TIME + 2
56   
57               lap_times.append(f"{code},{uniform(min_lap_time, max_lap_time):.3f}")
58   
59       shuffle(lap_times)
60   
61       with open(OUTPUT_FILE, "w") as out:
62           for lap_time in lap_times:
63               out.write(lap_time + "\n")
64   
65   
66   if __name__ == "__main__":
67       generate_the_lap_times()
