4    def format_and_sort_records(records):
5        from operator import itemgetter
6    
7        output = []
8    
9        for record in sorted(records, key=itemgetter(2), reverse=True):
10           first_name, surname, score = record
11           output.append(f"{surname:10} {first_name:10} {score:5.2f}")
12   
13       return "\n".join(output)
