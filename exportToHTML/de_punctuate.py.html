1    #!/usr/bin/env python3
2    
3    import sys
4    
5    
6    def remove_non_letters(s):
7        from string import punctuation
8    
9        return "".join([c for c in s if c not in punctuation])
10   
11   
12   def depunctuate_a_file(filename):
13       with open(filename, "r") as infile:
14           file_content = infile.read()
15   
16           with open(filename, "w") as outfile:
17               outfile.write(remove_non_letters(file_content))
18   
19   
20   if __name__ == "__main__":
21       try:
22           depunctuate_a_file(sys.argv[1])
23       except IndexError:
24           print(f"Usage: python3 {sys.argv[0]} <file to clean>")
25       except FileNotFoundError:
26           print(f'{sys.argv[0]}: Cannot open "{sys.argv[1]}" Sorry about that.')
