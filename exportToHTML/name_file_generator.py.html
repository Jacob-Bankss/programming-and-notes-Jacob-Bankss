1    #!/usr/bin/env python3
2    
3    
4    import names
5    from random import choice as pick, randint
6    from string import digits
7    import sys
8    
9    
10   NAMES_FILE = 'students.txt'
11   NUMBER_OF_NAMES = 128
12   UCAS_CODE_LENGTH = 7
13   UCAS_PREFIX = 'c'
14   
15   
16   def generate_ucas_code(prefix='c', length=8):
17       return prefix + ''.join([pick(digits) for _ in range(length)])
18   
19   
20   def generate_name():
21       forenames = randint(1, 3)
22       gender = 'male' if randint(1, 2) == 1 else 'female'
23   
24       name = names.get_last_name()
25       name += ', '
26       name += ' '.join([names.get_first_name(gender) for count in range(forenames)])
27   
28       return name
29   
30   
31   def generate_line():
32       return generate_ucas_code(UCAS_PREFIX, UCAS_CODE_LENGTH) + ' ' + generate_name()
33   
34   
35   if __name__ == '__main__':
36   
37       try:
38           out_file = sys.argv[1]
39       except IndexError:
40           out_file = NAMES_FILE
41   
42       with open(out_file, 'w') as out:
43           for line in range(NUMBER_OF_NAMES):
44               out.write(generate_line() + '\n')
