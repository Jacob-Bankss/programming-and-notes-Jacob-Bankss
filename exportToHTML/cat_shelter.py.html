7    def time_in_hours(total_minutes):
8        hours = total_minutes // 60
9        mins = total_minutes % 60
10   
11       hours_word = "Hour" if hours == 1 else "Hours"
12       minutes_word = "Minute" if mins == 1 else "Minutes"
13   
14       if hours > 0:
15           return f"{hours} {hours_word}, {mins} {minutes_word}"
16       else:
17           return f"{mins} {minutes_word}"
